'''
Scrivere un programma che
richiede in input un numero intero positivo
(se il numero non è positivo richiederlo nuovamente fino a quando risulti positivo)
visualizza tutti i suoi divisori
(suggerimento l'operatore % restituisce il resto della divisione)
'''
n = int(input('numero positivo: '))
while n <= 0:
    n = int(input('numero positivo: '))

print(f'divisori di {n}',end=' ⇒ ')
for d in range(1,n+1):
    if n % d == 0:
        print(d,end=' ')
print()
